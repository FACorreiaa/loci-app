Icon Generator

Open browser developer tools console and run the following script to generate icons:

// Create a 192x192 canvas
const canvas192 = document.createElement('canvas');
canvas192.width = 192;
canvas192.height = 192;
const ctx192 = canvas192.getContext('2d');

// Draw 192x192 icon
ctx192.fillStyle = '#007bff';
ctx192.fillRect(0, 0, 192, 192);
ctx192.fillStyle = 'white';
ctx192.font = 'bold 36px Arial';
ctx192.textAlign = 'center';
ctx192.textBaseline = 'middle';
ctx192.fillText('TUI', 96, 96);

// Create a 512x512 canvas
const canvas512 = document.createElement('canvas');
canvas512.width = 512;
canvas512.height = 512;
const ctx512 = canvas512.getContext('2d');

// Draw 512x512 icon
ctx512.fillStyle = '#007bff';
ctx512.fillRect(0, 0, 512, 512);
ctx512.fillStyle = 'white';
ctx512.font = 'bold 96px Arial';
ctx512.textAlign = 'center';
ctx512.textBaseline = 'middle';
ctx512.fillText('TUI', 256, 256);

// Download icons
function downloadCanvas(canvas, filename) {
    const link = document.createElement('a');
    link.download = filename;
    link.href = canvas.toDataURL('image/png');
    link.click();
}

downloadCanvas(canvas192, 'icon-192x192.png');
downloadCanvas(canvas512, 'icon-512x512.png');

console.log('Icons generated and downloaded!');
